import base64
from IPython.core.display import display, HTML

# Nombre del archivo de la imagen (ajustar segn el nombre real)
image_path = "jennie.jpg"

# Convertir la imagen a Base64 para incrustarla en el HTML
with open(image_path, "rb") as img_file:
    base64_image = base64.b64encode(img_file.read()).decode("utf-8")

# HTML con la imagen de fondo en Base64
html_code = f"""
<!DOCTYPE html>
<html lang="es">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Countdown - Jennie</title>
    <style>
        body {{
            font-family: Arial, sans-serif;
            background-color: #000;
            color: white;
            text-align: center;
            margin: 0;
            padding: 0;
            position: relative;
        }}
        .background {{
            position: fixed;
            top: 0;
            left: 0;
            width: 100%;
            height: 100%;
            background: url('data:image/jpeg;base64,{base64_image}') no-repeat center center;
            background-size: cover;
            z-index: -1;
        }}
        h1 {{
            font-size: 50px;
            margin-top: 20px;
            background: linear-gradient(to right, #ff0000, #ff7300);
            -webkit-background-clip: text;
            -webkit-text-fill-color: transparent;
            display: inline-block;
            padding: 10px 20px;
            border-radius: 10px;
        }}
        .countdown {{
            display: flex;
            justify-content: center;
            gap: 20px;
            margin-top: 20px;
        }}
        .box {{
            background: rgba(255, 255, 255, 0.8);
            color: red;
            padding: 20px;
            border-radius: 10px;
            width: 120px;
            text-align: center;
        }}
        .time {{
            font-size: 50px;
            font-weight: bold;
        }}
        .label {{
            font-size: 20px;
        }}
    </style>
</head>
<body>

    <div class="background"></div>  <!-- Imagen de fondo en Base64 -->
    <h1>COUNTDOWN</h1>
    <div class="countdown">
        <div class="box">
            <div id="days" class="time">00</div>
            <div class="label">Das</div>
        </div>
        <div class="box">
            <div id="hours" class="time">00</div>
            <div class="label">Horas</div>
        </div>
        <div class="box">
            <div id="minutes" class="time">00</div>
            <div class="label">Minutos</div>
        </div>
        <div class="box">
            <div id="seconds" class="time">00</div>
            <div class="label">Segundos</div>
        </div>
    </div>

    <script>
        // FECHA DEL LANZAMIENTO (AJUSTA SEGN TU EVENTO)
        const launchDate = new Date("March 6, 2025 23:00:00").getTime();

        function updateCountdown() {{
            const now = new Date().getTime();
            const timeLeft = launchDate - now;

            if (timeLeft <= 0) {{
                document.querySelector(".countdown").innerHTML = "<h2>El lbum de Jennie ya est disponible!</h2>";
                return;
            }}

            const days = Math.floor(timeLeft / (1000 * 60 * 60 * 24));
            const hours = Math.floor((timeLeft % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((timeLeft % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((timeLeft % (1000 * 60)) / 1000);

            document.getElementById("days").innerText = days.toString().padStart(2, "0");
            document.getElementById("hours").innerText = hours.toString().padStart(2, "0");
            document.getElementById("minutes").innerText = minutes.toString().padStart(2, "0");
            document.getElementById("seconds").innerText = seconds.toString().padStart(2, "0");
        }}

        setInterval(updateCountdown, 1000);
        updateCountdown(); // Para que no tarde 1 segundo en aparecer
    </script>

</body>
</html>
"""

# Mostrar en Google Colab
display(HTML(html_code))
